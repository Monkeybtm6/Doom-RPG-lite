ACTOR SPHolyFireItem : Inventory
{
	+FLOATBOB
	+INVENTORY.ALWAYSPICKUP
	inventory.maxamount 1
	Inventory.PickupMessage "Picked up Holy Fire" 
	States
	{
	Spawn:
		SPHF A -1 Bright
		Stop
	}
}

{
  //+INVENTORY.INVBAR
  +INVENTORY.AUTOACTIVATE
  +FLOATBOB
  +INVENTORY.ALWAYSPICKUP
  States
  {
  Spawn:
    SPHF A -1 Bright
    Stop
  Use:
    TNT1 A 0 A_JumpIfInventory("Mana",50,"ConfirmUse")
