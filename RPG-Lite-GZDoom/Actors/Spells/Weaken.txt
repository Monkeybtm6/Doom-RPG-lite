ACTOR SPWeaken : Inventory
{
	+FLOATBOB
	+INVENTORY.ALWAYSPICKUP
	inventory.maxamount 1
	Inventory.PickupMessage "Picked up Weaken" 
	States
	{
	Spawn:
		SPFR A -1 Bright
		Stop
	}
}

  +INVENTORY.ALWAYSPICKUP

}
